float getamount(float currentbal) {
    float amnt;
    printf("Enter the amount to withdraw: ");
    scanf("%f", &amnt);
    if (amnt > currentbal) {
        printf("Insufficient balance Current balance: %f\n", currentbal);
        return -1;
    }
    return amnt;
}
